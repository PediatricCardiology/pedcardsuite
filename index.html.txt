    <script>
        let readings = [];
        let chartInstance = null;

        function addReading() {
            const dateVal = document.getElementById('dateInput').value;
            const inrVal = parseFloat(document.getElementById('inrInput').value);

            if (!dateVal || isNaN(inrVal)) return;

            readings.push({ date: dateVal, value: inrVal, id: Date.now() });
            readings.sort((a, b) => new Date(a.date) - new Date(b.date));
            
            // Clear inputs
            document.getElementById('inrInput').value = '';
            
            updateCalculations();
            renderHistory();
        }

        function removeReading(id) {
            readings = readings.filter(r => r.id !== id);
            updateCalculations();
            renderHistory();
        }

        function renderHistory() {
            const list = document.getElementById('historyList');
            if (readings.length === 0) {
                list.innerHTML = `<p class="text-xs text-slate-400 italic text-center py-4">No data entered yet.</p>`;
                return;
            }

            list.innerHTML = readings.map(r => `
                <div class="flex items-center justify-between p-3 bg-white border border-slate-100 rounded-xl group transition-all hover:border-blue-200">
                    <div class="text-xs">
                        <div class="font-black text-slate-700">${r.date}</div>
                        <div class="text-slate-400 font-bold uppercase tracking-tighter" style="font-size: 8px;">Measured Reading</div>
                    </div>
                    <div class="flex items-center gap-4">
                        <span class="text-sm font-black text-blue-600">${r.value.toFixed(1)}</span>
                        <button onclick="removeReading(${r.id})" class="text-slate-200 hover:text-red-500 transition-colors">
                            <svg xmlns="http://www.w3.org/2000/svg" class="h-4 w-4" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16" />
                            </svg>
                        </button>
                    </div>
                </div>
            `).join('');
        }

        function updateCalculations() {
            const lower = parseFloat(document.getElementById('lowerLimit').value);
            const upper = parseFloat(document.getElementById('upperLimit').value);

            if (readings.length === 0) {
                document.getElementById('rosendaalVal').innerText = '--';
                document.getElementById('traditionalVal').innerText = '--';
                updateChart([], lower, upper);
                return;
            }

            // Traditional Calculation
            const inRangeCount = readings.filter(r => r.value >= lower && r.value <= upper).length;
            document.getElementById('traditionalVal').innerText = ((inRangeCount / readings.length) * 100).toFixed(1);

            // Rosendaal Calculation
            if (readings.length < 2) {
                document.getElementById('rosendaalVal').innerText = '--';
            } else {
                let totalDays = 0;
                let daysInRange = 0;

                for (let i = 0; i < readings.length - 1; i++) {
                    const start = readings[i];
                    const end = readings[i+1];
                    const diffTime = Math.abs(new Date(end.date) - new Date(start.date));
                    const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));

                    if (diffDays <= 0) continue;

                    const dailyChange = (end.value - start.value) / diffDays;

                    for (let d = 0; d < diffDays; d++) {
                        const interpolated = start.value + (dailyChange * d);
                        if (interpolated >= lower && interpolated <= upper) daysInRange++;
                        totalDays++;
                    }
                }
                // Check last reading
                const last = readings[readings.length - 1];
                if (last.value >= lower && last.value <= upper) daysInRange++;
                totalDays++;

                document.getElementById('rosendaalVal').innerText = ((daysInRange / totalDays) * 100).toFixed(1);
            }

            updateChart(readings, lower, upper);
        }

        function updateChart(data, lower, upper) {
            const ctx = document.getElementById('ttrChart').getContext('2d');
            
            if (chartInstance) chartInstance.destroy();

            chartInstance = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: data.map(r => r.date),
                    datasets: [{
                        label: 'INR Reading',
                        data: data.map(r => r.value),
                        borderColor: '#2563eb',
                        borderWidth: 4,
                        backgroundColor: 'rgba(37, 99, 235, 0.1)',
                        fill: true,
                        tension: 0.3,
                        pointBackgroundColor: '#fff',
                        pointBorderColor: '#2563eb',
                        pointBorderWidth: 2,
                        pointRadius: 5
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: { display: false },
                        tooltip: {
                            backgroundColor: '#1e293b',
                            padding: 12,
                            titleFont: { size: 10 },
                            bodyFont: { weight: 'bold' },
                            cornerRadius: 12
                        }
                    },
                    scales: {
                        y: {
                            beginAtZero: false,
                            min: 1,
                            max: 5,
                            grid: { color: '#f1f5f9' },
                            ticks: { font: { size: 10, weight: 'bold' } }
                        },
                        x: {
                            grid: { display: false },
                            ticks: { font: { size: 9, weight: 'bold' } }
                        }
                    }
                },
                plugins: [{
                    id: 'highlightRange',
                    beforeDraw: (chart) => {
                        const { ctx, chartArea, scales: { y } } = chart;
                        const yStart = y.getPixelForValue(upper);
                        const yEnd = y.getPixelForValue(lower);
                        ctx.save();
                        ctx.fillStyle = 'rgba(16, 185, 129, 0.08)';
                        ctx.fillRect(chartArea.left, yStart, chartArea.width, yEnd - yStart);
                        ctx.restore();
                    }
                }]
            });
        }

        // Initialize with empty chart
        updateChart([], 2.0, 3.0);
    </script>
